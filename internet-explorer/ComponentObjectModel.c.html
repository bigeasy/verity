BOOLEAN
IsIIDImplemented(REFIID riid, REFIID ariidImplemented[])
{
    DWORD i;
    for (i = 0; ariidImplemented[i]; i++)
    {
        if (IsEqualIID(riid, ariidImplemented[i]))
        {
            return TRUE;
        }
    }
    return FALSE;
}
